class EncryptionSystem:
    def __init__(self):
        self.key = Fernet.generate_key()
        self.cipher = Fernet(self.key)
        
    def encrypt_data(self, data: str) -> str:
        encrypted = self.cipher.encrypt(data.encode())
        return base64.urlsafe_b64encode(encrypted).decode()
    
    def decrypt_data(self, encrypted_data: str) -> str:
        decoded = base64.urlsafe_b64decode(encrypted_data)
        return self.cipher.decrypt(decoded).decode()

class BotDetectionSystem:
    def __init__(self):
        self.bot_patterns = [
            r'@.*bot$',  # –ó–∞–≤–µ—Ä—à—É—î—Ç—å—Å—è –Ω–∞ bot
            r'Bot$',     # –ó–∞–≤–µ—Ä—à—É—î—Ç—å—Å—è –Ω–∞ Bot
            r'–±–æ—Ç$',     # –ó–∞–≤–µ—Ä—à—É—î—Ç—å—Å—è –Ω–∞ –±–æ—Ç
        ]
        
    async def analyze_user(self, user: User) -> Dict:
        indicators = {
            'is_bot': user.bot if hasattr(user, 'bot') else False,
            'username_pattern': any(re.match(p, user.username or '') for p in self.bot_patterns),
            'no_profile_photo': user.photo is None,
            'recent_creation': self.is_recent_account(user),
            'suspicious_bio': self.check_suspicious_bio(user),
        }
        
        return indicators
    
    def is_recent_account(self, user: User) -> bool:
        # –ü–µ—Ä–µ–≤—ñ—Ä–∫–∞ —á–∏ –∞–∫–∫–∞—É–Ω—Ç —Å—Ç–≤–æ—Ä–µ–Ω–∏–π –Ω–µ—â–æ–¥–∞–≤–Ω–æ
        if hasattr(user, 'status'):
            if hasattr(user.status, 'was_online'):
                return user.status.was_online > datetime.now() - timedelta(days=7)
        return False
    
    def check_suspicious_bio(self, user: User) -> bool:
        suspicious_keywords = ['crypt', 'VPN', 'proxy', 'spoof', 'clone']
        bio = (user.about or '').lower()
        return any(word in bio for word in suspicious_keywords)
